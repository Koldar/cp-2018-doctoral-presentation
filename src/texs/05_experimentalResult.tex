\section{Experiments}

\subsection{}
\begin{frame}{Benchmarks}

	\begin{itemize}
		\item Randomly generated inconsistent CSPs over PA and ORD-Horn (either \textit{sparse} or \textit{dense});
		\item inconsistent CSPs over PA generated from ZenoTravel planning instances (plans);
		\item relaxation policies:
			\begin{itemize}
				\item \textbf{RR}: random relaxations of a randomly chosen constraint;
				\item \textbf{RFSCC}: random relaxations of a constraint between time variables in the same flawed SCC.
			\end{itemize}
	\end{itemize}

\end{frame}
